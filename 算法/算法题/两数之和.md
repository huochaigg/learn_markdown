
获取数组中 两个相加能等于target数字的索引

```
function getTwoSumIndex(arr, target) {
	const targetMap = new Map();
	for (let i in arr) {
	const curDiff = target - arr[i]
	if (targetMap.has(curDiff)) {
		return [targetMap.get(curDiff), i]
	}
		targetMap.set(arr[i], i)
	}
}

const numArr = [2, 7, 11, 15];
const result = getTwoSumIndex(numArr, 9);
console.log('result:', result); // [0, 1]
```
